#| label: fig-submissions-by-day
#| fig-cap: Submissions by days until deadline. 

# define key dates
deadline <- as.Date("2025-11-11") 
opening  <- as.Date(deadline - weeks(4))         


# compute days until deadline
submissions <- submissions |> 
  mutate(days_until_deadline = as.numeric(difftime(deadline, day, units = "days")))

# define palette
wes_cols <- wes_palette("Zissou1", 5, type = "discrete")
blue   <- wes_cols[1]  # distribution
yellow <- wes_cols[3]  # opening
red    <- wes_cols[5]  # deadline

# Create plot
submissions |> 
  count(days_until_deadline) |> 
  mutate(prop = n / sum(n)) |> 
  ggplot(aes(x = days_until_deadline, y = n)) +
  geom_col(fill = blue) +
  geom_vline(xintercept = 0, color = red, linetype = "dashed", linewidth = 1) +
  geom_vline(xintercept = as.numeric(difftime(deadline, opening, units = "days")), 
             color = yellow, linetype = "dashed", linewidth = 1) +
  annotate("text", x = 0, y = 60, label = "Deadline", color = red, hjust = 1.4, size = 3.5) +
  annotate("text", 
           x = as.numeric(difftime(deadline, opening, units = "days")), 
           y = 40, label = "Opening", color = yellow, hjust = -0.2, size = 3.5) +
  scale_x_reverse(breaks = seq(0, 28, by = 2)) +
  labs(
    x = "Days until deadline",
    y = "Number of submissions"
  ) +
  plot_theme
